// Example 1: Complex App Filtering
suspend fun getConfiguredApps(): List<AppEntity> {
    return appDao.getAllApps().filter { app ->
        app.isEnabled && app.customSoundUri != null
    }.sortedByDescending { it.lastUsed }
}
